CREATE TABLE provider_locations (
    uniqid INT AUTO_INCREMENT PRIMARY KEY,
    provider_location_guid varchar(255),
    loc_admin_street1 varchar(100),
    tuesday_hours varchar(100),
    thursday_hours varchar(100),
    sunday_hours varchar(100),
    searchable_name varchar(255),
    saturday_hours varchar(100),
    monday_hours varchar(100),
    friday_hours varchar(100),
    loc_store_no varchar(100),
    wednesday_hours varchar(100),
    web_address varchar(50),
    pre_screen varchar(50),
    category varchar(255),
    loc_admin_street2 varchar(100),
    loc_admin_city varchar(100),
    loc_phone varchar(100),
    loc_name varchar(100),
    loc_admin_state varchar(2),
    loc_admin_zip varchar(10),
    insurance_accepted bool,
    in_stock bool, 
    walkins_accepted bool,
    provider_notes text,
    supply_level int,
    geopoint point,
    quantity_last_updated date
);
